63  {
64  Degree = degree;
65  // Pass 1: Determine the current state and the next element
66  for (int i = 0; i < dataset.Length () - degree; i++) {
67  var state = Transformations.Subsequence (dataset, i, degree).ToArray ();
68  var future = dataset.Get (i + degree);
69  States.Add(
70  new MarkovState<T>(
71  state,
72  future));
73  }

75  // Pass 2: Determine the probability of current incurring future state
76  foreach (MarkovState<T> ms in States) {
77  List<T> occurences = (from MarkovState<T> item in States
78  where Information.Equal(item.State, ms.State)
79  select item.Next).ToList();
80  // Probability is equal to the times this future state occured compared to how many there are.
81  ms.Probability = ((double)Information.Times<T> (occurences, ms.Next)) / ((double)occurences.Length<T> ());
82  }
